numbers = [1, 10, 0, 4, 9]


def bubble_sort(array):
    size = len(array)
    for i in range(size):
        for j in range(size - 1):
            if array[j] > array[j+1]:
                item = array[j]
                array[j] = array[j+1]
                array[j + 1] = item

    return array


print(bubble_sort(numbers))
